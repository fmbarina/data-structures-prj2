43  /* Compute the byte frequency within the file */
44  tCounter *bytes_freq_count = counter_init();
45  unsigned char c_byte;
46  while (regl_read_single_byte(input, &c_byte) == 1)
47  counter_byte_inc(bytes_freq_count, (unsigned char)c_byte);
48 
49  regl_reset_cursor(input);
50

52  int i;
53  tTreeList *tree_list = tree_list_inic();
54  for (i = 0; i < 256; i++)
55  {
56  unsigned long long int freq = counter_get_vetx(bytes_freq_count, i);
57  if (freq == 0)
58  continue;
59  tree_list_insert(tree_list, tree_init(i, freq, NULL, NULL));
60  }

63  tTree *optimized_tree = huffman_algorithm(tree_list);
64 
65  /* Prepare compacted file data for writing */
66  comp_set_name(output, regl_get_name(input));
67  comp_set_ext(output, regl_get_ext(input));
68  comp_set_tree(output, optimized_tree);
69

71  comp_write_header(output);

73  /* Max num. of bytes we can decode at a time */
74  unsigned int enc_lim_in = encode_limit_in();
75 
76  /* Max num. of bits we can expect out of decoding */
77  unsigned int enc_lim_out = encode_limit_out();
78 
79  /* Max num. of bits we can write to the compacted file at a time */
80  unsigned int dec_lim_in = decode_limit_in();
81 
82  /* Compress and write file contents */
83  while (1)
84  {
85  tBytes *read_buf = bytes_init(enc_lim_in);
86  tBits *encode_buf = bits_init(enc_lim_out);
87 
88  /* Read up to numBytes from input file, save the return in r_ret */
89  unsigned int r_ret = regl_read_bytes(input, read_buf);
90 
91  /* If there are no bytes (file has ended or can't be read), stop */
92  if(!r_ret)
93  {
94  bytes_free(read_buf);
95  bits_free(encode_buf);
96  break;
97  }
98 
99  /* Get encoded tBits with the read bytes using the encoding tree */
100  encode(comp_get_tree(output), read_buf, encode_buf);
101 
102  /* Due to size constraints when decompacting the file later, we should
103  * avoid trying to write too many bits at once to the compacted file.
104  * Instead, we will divide the encoded buffer in batches of size
105  * 'dec_lim_in' and write them, one batch at a time.
106  */
107  unsigned int encode_len = bits_get_len(encode_buf);
108  unsigned int nsets = ((encode_len - 1) / dec_lim_in);
109  unsigned int batch;
110  for (batch = 0; batch <= nsets; batch++)
111  {
112  tBits *write_buf = bits_init(dec_lim_in);
113 
114  /* Fill write buffer with bits from the set */
115  unsigned int nbits = (batch == nsets) ? (encode_len % dec_lim_in) : (dec_lim_in);
116  unsigned int bit;
117  for (bit = 0; bit < nbits; bit++)
118  {
119  unsigned int index = (batch * dec_lim_in) + bit;
120  bits_append(write_buf, bits_get_bit(encode_buf, index));
121  }
122 
123  /* Send the buffer to be written */
124  comp_write_data(output, write_buf);
125 
126  /* Reset the buffer to start the next set */
127  bits_free(write_buf);
128  }
129 
130  /* Free the buffers to start the next loop clear */
131  bytes_free(read_buf);
132  bits_free(encode_buf);
133  }
134 
135  /* Free memory and close files */
136  counter_free(bytes_freq_count);
137  tree_list_free(tree_list);
138  regl_free(input);
139  comp_free(output);
140 
141  return 0;
142 }
