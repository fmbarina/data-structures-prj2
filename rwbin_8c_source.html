123 {

126

128 
129  if (size[0] > 0)
130  {
131  unsigned int bytes = ((size[0] + 7) / 8);
132 
133  unsigned char *values = (unsigned char *)
134  calloc(bytes, sizeof(unsigned char));
135 
136  ret = fread(values, sizeof(unsigned char), bytes, f);
137 
138  unsigned int i;
139  for (i = 0; i < size[0]; i++)

141  unsigned char val = values[i / 8];

144
